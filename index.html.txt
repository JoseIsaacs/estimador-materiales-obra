<h2>Estimador de Bloques</h2>

<label>Área de pared (m²)</label>
<input id="pared">

<button onclick="bloques()">Calcular Bloques</button>

<div id="resultado_bloques"></div>
<script>

function calcular(){

let area = document.getElementById("area").value;
let espesor = document.getElementById("espesor").value;

let volumen = area * (espesor/100);

document.getElementById("resultado").innerHTML =
"Concreto necesario: " + volumen.toFixed(2) + " m³";

}
function bloques(){

let pared = document.getElementById("pared").value;

let bloques = pared * 12.5;

let mortero = pared * 0.02;

document.getElementById("resultado_bloques").innerHTML =
"Bloques necesarios: " + Math.ceil(bloques) +
"<br>Mortero aproximado: " + mortero.toFixed(2) + " m³";

}

<h2>Estimador de Baldosas</h2>

<label>Área del piso (m²)</label>
<input id="area_piso">

<label>Tamaño baldosa (cm)</label>
<input id="lado_baldosa">

<button onclick="baldosas()">Calcular Baldosas</button>

<div id="resultado_baldosas"></div>

function baldosas(){

let area = document.getElementById("area_piso").value;
let lado = document.getElementById("lado_baldosa").value;

let area_baldosa = (lado/100) * (lado/100);

let cantidad = area / area_baldosa;

cantidad = cantidad * 1.1;

document.getElementById("resultado_baldosas").innerHTML =
"Baldosas necesarias (incluye 10% desperdicio): " + Math.ceil(cantidad);

}

<h2>Columnas de Concreto</h2>

<label>Cantidad de columnas</label>
<input id="num_col">

<label>Altura (m)</label>
<input id="altura_col">

<label>Sección (cm)</label>
<input id="seccion_col">

<button onclick="columnas()">Calcular</button>

<div id="resultado_columnas"></div>
function columnas(){

let n = document.getElementById("num_col").value;
let h = document.getElementById("altura_col").value;
let s = document.getElementById("seccion_col").value;

let area = (s/100)*(s/100);

let volumen = area * h * n;

let acero = h * n * 4;

document.getElementById("resultado_columnas").innerHTML =
"Volumen concreto: " + volumen.toFixed(2) + " m³" +
"<br>Acero longitudinal aprox: " + acero.toFixed(1) + " m";

}
