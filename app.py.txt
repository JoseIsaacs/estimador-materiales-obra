# Sistema de Estimación de Materiales

print("Estimador de Materiales para Obras")

largo = float(input("Ingrese largo del muro (m): "))
alto = float(input("Ingrese alto del muro (m): "))

area = largo * alto
bloques_por_m2 = 12.5
bloques = area * bloques_por_m2

print("Área del muro:", area, "m2")
print("Bloques estimados:", round(bloques))
